#### 문제 아이디어

기존 펠린드롬 함수를 사용하면서 문자열을 하나 뺐을때도 펠린드롬이 되는지 만 확인하는 문제라고 생각했다. 문자열 탐색할때는 시간이 부족할 것 같아서 이진탐색을 이용해서 문자열을 탐색해야 겠다고 생각했다 .

---

#### 코드 

```js
let fs = require('fs')
let input = fs.readFileSync('/dev/stdin').toString().split("\n")

let n = Number(input[0])

function palindrome(str) {
  return str == str.split('').reverse().join('') 
}

for (let i=1; i<=n ;i++){
  let str = input[i]

  let left = 0;
  let right = str.length-1;
  let flag = false;

  if (palindrome(str)){
    console.log(0)
    continue
  }
  
  while (left < right) {
    if (str[left] == str[right]){
      left ++;
      right --;
    } else {
      if (palindrome(str.slice(left+1, right+1)) || palindrome(str.slice(left, right))){
        flag = true 
        break
      } else {
        break
      }
    }
  }

  if (flag) {
    console.log(1)
  } else {
    console.log(2)
  }
}
```
---

#### 코드 해석

유사 회문 인지 아닌지 판단하기 위해서 이진탐색을하면서 문자열이 다를 시에 문자열을 왼쪽 부분 1개 또는 오른쪽 부분 1개 자른 것이 펠린드롬이 아닌지 판단하도록 했다 .

---

#### 어려웠던 점

문자열을 시간내에 탐색해야 했기에 이진탐색을 어떻게 구현해야 하는지 조금 애를 먹었다.
