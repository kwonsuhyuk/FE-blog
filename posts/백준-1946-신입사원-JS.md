#### 문제 아이디어

arr 에 각 데이터들을 담아서 각 데이터의 첫번째 데이터로 정렬 후에 첫번째 두번째 최솟값을 크게 잡고
배열을 돌려서 둘다 작을 시에는 카운트를 안하고 둘중 하나라도 작을 시에는 최솟값을 교체해 주는 식으로 아이디어를 냈다. 

#### 코드 

```js
let fs = require('fs')
let input = fs.readFileSync('/dev/stdin').toString().split("\n")

let t = Number(input[0])
let line = 1

for (let i=0; i<t;i++){
  let n = Number(input[line])
  let arr = []
  for (let j=line+1; j<=n+line; j++){
    let data = input[j].split(' ').map(Number)
    arr.push(data)
    
  }
  arr.sort((a,b) => a[0] - b[0])
  
  let minA = 100001
  let minB = 100001


  let res = 0
  
  for ([a,b] of arr){
    if (a >= minA && b >= minB) {
      continue
    } else{
      minA = Math.min(a,minA)
      minB = Math.min(b,minB)
      res ++
    }
   
  }
  console.log(res) 
  line += n+1
  
}
```

#### 코드 해석 

배열을 돌면서 a,b 값을 추출 후에 `Math.min()` 두개의 인수 중 작은 것을 넣어주고 카운트를 하였다.
