#### 문제 아이디어

문자열을 -로 나누어서 그 안의 수를 다 더해서 빼는 방식으로 구현하였다.

#### 처음시도한 코드(실패)
```js
let fs = require('fs')
let input = fs.readFileSync('/dev/stdin').toString().split("\n")

let str = input[0]

if (str.includes('-')){
  let arr = str.split('-')
  let res = Number(arr[0])
  for (let i=1; i<arr.length; i++){
    let tmp = arr[i].split('+').map(Number).reduce((acc,cur)=> acc+cur,0)
    res -= tmp
  }
  console.log(res)
} else {
  let arr = str.split('+').map(Number)
  let res = arr.reduce((acc,cur)=> acc+cur, 0)
  console.log(res)
}
```
- 첫번째 항을 잘못 처리하는 것 같아 모든항을 동일하게 처리하는 식으로 수정해야 겠다고 생각이 들었다.



#### 수정코드
```js
let fs = require('fs')
let input = fs.readFileSync('/dev/stdin').toString().split("\n")

let str = input[0].split('-')

let res = 0

for (let i=0; i<str.length; i++){
  
  let tmp = str[i].split('+').map(Number).reduce((a,b)=>a+b,0)
  if (i == 0) res+= tmp
  else res -= tmp
}
console.log(res)
```

### 코드해석

`let str = input[0].split('-')` 

- -부호 기준으로 나눈후에 그 안의 숫자들은 + 부호 일테니 모두 더해서 첫번째 숫자일시에는 더해주고 나머지일시에는 모두 빼준다.
